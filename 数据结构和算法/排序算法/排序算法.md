## 归并排序

```js
let arr = [4, 6, 5, 2, 1, 7, 3];
mergeSort(arr);

function mergeSort(array) {
  mergeSortPath(array, 0, array.length-1);
}

function mergeSortPath(array, p, r) {
  if(p >= r) return;
  
  let q = Math.floor(( p + r ) / 2); // 分割点
    
  mergeSortPath(array, p, q);
  mergeSortPath(array, q+1, r);
  
  merge(array, p, q, r); // 合并排序后的两个部分
}

function merge(array, p, q, r) {
  let tmp = [];
  let i = p; 
  let j = q + 1;
  
  while(i <= q && j <= r){
    if(array[i] <= array[j]){
      tmp.push(array[i++]);
    } else{
      tmp.push(array[j++]);
    }
  }
  
  let start,end;
  start = i != q + 1 ? i : j;
  end = i != q + 1 ? q : r;
  
  while (start <= end) {
    tmp.push(array[start++]);
  }
  
  tmp.forEach((item, index) => {
    array[p+index] = item;
  })
}
```
